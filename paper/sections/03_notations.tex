\section{Notations}
\label{sec:notations}

For clarity and consistency throughout this paper, we summarize the key symbols used in our models in Table~\ref{tab:notations}.

\begin{table}[H]
    \centering
    \caption{Summary of Key Notations}
    \label{tab:notations}
    \begin{tabularx}{\textwidth}{c X c}
        \toprule
        \textbf{Symbol} & \textbf{Description} & \textbf{Unit/Range} \\
        \midrule
        $s$ & Season index & $s \in \{1,\ldots,34\}$ \\
        $t$ & Week index within a season & $t \in \{1,\ldots,T_s\}$ \\
        $i$ & Contestant index & -- \\
        $\mathcal{C}_{s,t}$ & Active contestant set in season $s$, week $t$ & -- \\
        $n_{s,t}$ & Number of active contestants, $|\mathcal{C}_{s,t}|$ & -- \\
        \midrule
        $J_{i,t}$ & Judge score sum for contestant $i$ in week $t$ & [0, 40] \\
        $Jpct_{i,t}$ & Judge percent, $J_{i,t}/\sum_k J_{k,t}$ & $[0,1]$ \\
        $E_t$ & Eliminated contestant index in week $t$ & -- \\
        $r^{\mathrm{fan}}_{i,t}$ & Fan vote rank (latent in rank seasons) & $\{1,\ldots,n_{s,t}\}$ \\
        \midrule
        $v_{i,t}$ & Fan support share for contestant $i$ in week $t$ & $[0,1]$, $\sum_i v_{i,t}=1$ \\
        $\hat{v}_{i,t}$ & Posterior mean of fan support share & $[0,1]$ \\
        $[v^{\min}_{i,t}, v^{\max}_{i,t}]$ & Feasible interval from LP/MILP & $[0,1]$ \\
        $\mathrm{HDI}_{i,t}$ & 95\% highest density interval for $v_{i,t}$ & $[0,1]$ \\
        \midrule
        $\epsilon$ & Minimum vote share floor & $[0,0.1]$ \\
        $S^*$ & Minimum slack (Assumption--Data Tension) & $\ge 0$ \\
        $\lambda$ & Temporal smoothness penalty in MCMC & $>0$ \\
        $\alpha$ & Judge weight in hybrid mechanism & $[0,1]$ \\
        \midrule
        $T_i$ & Survival weeks of contestant $i$ & weeks \\
        $f(\mathbf{x}_i)$ & XGBoost prediction for $T_i$ & weeks \\
        $\phi_j$ & SHAP value for feature $j$ & -- \\
        \midrule
        $Jpct_{i,t}$ & Judge percentage for contestant $i$ & $[0,1]$ \\
        $S_{i,t}^{\mathrm{wp}}$ & Weighted-percent score, $\alpha Jpct_{i,t} + (1-\alpha)v_{i,t}$ & $[0,1]$ \\
        \bottomrule
    \end{tabularx}
\end{table}

\noindent\textbf{Indexing Conventions.}
\begin{itemize}[itemsep=0.2em]
    \item Seasons are numbered chronologically: S1 (2005) through S34 (2025).
    \item Weeks $t$ are numbered starting from the first competition episode (excluding any premiere without elimination).
    \item \emph{Percent-rule seasons} (S3--S27): combined scores defined by percentage rule; fan shares unobserved.
    \item \emph{Rank-rule seasons} (S1--S2, S28--S34): only ordinal rankings are revealed.
\end{itemize}
