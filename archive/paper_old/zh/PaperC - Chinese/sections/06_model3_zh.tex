\subsection{DAWS 动态权重机制}

\begin{equation}
\alpha_t = \mathrm{clip}\Big(\alpha_0 + \gamma \frac{t}{T} - \eta U_t,\ \alpha_{\min},\alpha_{\max}\Big), \quad
|\alpha_t-\alpha_{t-1}|\le \delta,
\end{equation}
其中 $U_t$ 为不确定性代理。
